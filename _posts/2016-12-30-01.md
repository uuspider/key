### 迭代法

    #!/usr/bin/env python
    import math
    x0 = 1.3
    def eq(x):
        x1 = 2-math.log(x)
        return x1
    x1 = eq(x0)
    i = 0
    while math.fabs(x0-x1) > 0.00000000001:
        i+=1
        x0 = x1
        x1 = eq(x0)
    print x1
    print i
