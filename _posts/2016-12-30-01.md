### 迭代法

    #!/usr/bin/env python
    import math
    x0 = 1.3
    def eq(x):
        x1 = 2-math.log(x)
        return x1
    x1 = eq(x0)
    i = 0
    while math.fabs(x0-x1) > 0.00000000001:
        i += 1
        x0 = x1
        x1 = eq(x0)
    print x1

### 加速迭代法

    #!/usr/bin/env python
    import math
    x0 = 1.3
    def eq(x):
        x1 = 2-math.log(x)
        return x1
    x1 = eq(x0)
    x2 = eq(x1)
    i = 0
    while math.fabs(x2-x1) > 0.00000000001:
        if x2-2*x1+x0 == 0:
            break
        i += 1
        x0 = x2-((x2-x1)**2)/(x2-2*x1+x0)
        x1 = eq(x0)
        x2 = eq(x1)
    print x2
    print i
